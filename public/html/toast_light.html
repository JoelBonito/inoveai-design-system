UsageExample.tsx
import { useToast } from '@devstudio/ui';

export const DeploymentAction = () => {
  const { toast } = useToast();

  const handleDeploy = async () => {
    try {
      await deployProject();
      
      // Success State
      toast.success({
        title: 'Deployment Successful',
        description: 'Project Alpha is live.',
        duration: 5000,
      });

    } catch (error) {
      // Error State
      toast.error({
        title: 'Compilation Failed',
        description: 'Syntax error in main.ts'
      });
    }
  };
  
  return (
    <Button onClick={handleDeploy}>
      Deploy
    </Button>
  );
};
