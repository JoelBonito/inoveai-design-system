Usage

import { Select } from '@nexus/ui';

export default function ModelSelector() {
  return (
    <Select
      label="AI Model"
      placeholder="Select inference engine..."
      size="md"
      options={[
        { label: 'GPT-4', value: 'gpt4' },
        { label: 'Claude 3', value: 'claude3' },
      ]}
      onChange={(val) => console.log(val)}
    />
  );
}
