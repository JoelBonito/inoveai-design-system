CreateProject.tsx 
import { Select, SelectItem } from '@agency/ui';

export function ModelConfig() {
  return (
    <Select 
      label="AI Model" 
      placeholder="Pick a model..."
      defaultValue="gpt-4"
    >
      <SelectItem value="gpt-4">
        GPT-4 Turbo
      </SelectItem>
      <SelectItem value="claude-3">
        Claude 3 Opus
      </SelectItem>
    </Select>
  );
}
