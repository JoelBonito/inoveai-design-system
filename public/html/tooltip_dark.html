Installation & Usage

MyComponent.tsx 
import { Tooltip } from '@agency/ui-kit';

export const AgentStatus = () => {
  return (
    <Tooltip
      content={
        <div className="flex flex-col">
           <span>Reasoning engine initialized</span>
           <span className="text-gray-400">Latency: 24ms</span>
        </div>
      }
      placement="top"
      delay={200}
    >
      <Button variant="primary">
         Initialize Agent
      </Button>
    </Tooltip>
  );
};
