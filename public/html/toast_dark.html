Implementation

App.tsx 
import { useToast } from '@/hooks/use-toast'
import { Button } from '@/components/ui/button'

export default function ToastDemo() {
  const { toast } = useToast()

  return (
    <Button
      onClick={() => {
        toast({
          title: "Deployment Successful",
          description: "Project 'Nebula' has been deployed.",
          variant: "success",
        })
      }}
    >
      Show Toast
    </Button>
  )
}
