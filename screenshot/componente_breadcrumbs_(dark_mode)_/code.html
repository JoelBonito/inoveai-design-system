Usage

example.tsx 
import { Breadcrumb, BreadcrumbItem, BreadcrumbLink, BreadcrumbSeparator } from "@ai-sys/ui"

export function ProjectHeader() {
  return (
    <Breadcrumb>
      <BreadcrumbItem>
        <BreadcrumbLink href="/">Home</BreadcrumbLink>
      </BreadcrumbItem>
      <BreadcrumbSeparator />
      <BreadcrumbItem>
        <BreadcrumbLink href="/components">Components</BreadcrumbLink>
      </BreadcrumbItem>
      <BreadcrumbSeparator />
      <BreadcrumbItem>
        <BreadcrumbPage>Breadcrumb</BreadcrumbPage>
      </BreadcrumbItem>
    </Breadcrumb>
  )
}

API Reference

Prop Type Default Description
separator ReactNode <ChevronRight /> Custom separator element between items.
addHomeIcon boolean false If true, adds a default home icon to the first item.
maxItems number undefined Maximum number of items to show before truncating.
variant 'default' | 'mono' 'default' Visual style of the breadcrumb trail.
