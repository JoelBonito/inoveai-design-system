Usage

React / Typescript Copy 
import { Badge } from '@agency/ui';

export default function TransactionStatus({ status }) {
  if (status === 'success') {
    return (
      <Badge variant="soft" color="success" icon="check">
        Payment Verified
      </Badge>
    );
  }
  
  return <Badge variant="outline">Processing</Badge>;
}
