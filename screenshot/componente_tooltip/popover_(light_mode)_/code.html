Usage

MyComponent.tsx 
import { Tooltip, TooltipContent, TooltipTrigger } from "@ai-design/ui";

export function AnalysisButton() {
  return (
    <Tooltip delayDuration={300}>
      <TooltipTrigger asChild>
        <Button variant="outline">Hover me</Button>
      </TooltipTrigger>
      <TooltipContent side="top">
        <p>AI Confidence Score: 98%</p>
      </TooltipContent>
    </Tooltip>
  );
}

API Reference

Prop Type Default Description
content ReactNode | string - The content to display inside the popover/tooltip.
side "top" | "right" | "bottom" | "left" "top" The preferred side of the trigger to render against.
sideOffset number 4 The distance in pixels from the trigger.
delayDuration number 700 The duration from when the mouse enters the trigger until the tooltip opens.
